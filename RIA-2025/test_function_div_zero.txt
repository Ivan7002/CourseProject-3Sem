function integer divide(integer a, integer b) {
    declare integer res;
    res = a / b;  // Деление на параметр b
    return res;
}

function bool isEven(integer v) {
    return (v % 2) == 0;  // Деление на параметр v (остаток от деления)
}

main {
    declare integer x;
    declare integer e;

    x = 10;
    e = 0;

    cout "Testing division by zero in function calls:\n";

    cout "divide(10, 5) = "; cout divide(10, 5); cout "\n";  // OK
    cout "divide(10, 0) = "; cout divide(10, 0); cout "\n";  // ДОЛЖНА ЛОВИТЬСЯ
    cout "divide(x, 0) = "; cout divide(x, 0); cout "\n";    // ДОЛЖНА ЛОВИТЬСЯ
    cout "divide(x, e) = "; cout divide(x, e); cout "\n";    // ДОЛЖНА ЛОВИТЬСЯ (e = 0)

    cout "isEven(4) = "; cout isEven(4); cout "\n";          // OK
    cout "isEven(0) = "; cout isEven(0); cout "\n";          // ДОЛЖНА ЛОВИТЬСЯ
}
